给定一个数组，它的第 i 个元素是一支给定股票第 i 天的价格。

如果你最多只允许完成一笔交易（即买入和卖出一支股票），并且只能在一个单调增区间内进行买卖，设计一个算法来计算你所能获取的最大利润。

注意你不能在买入股票前卖出股票。



```java
package com.xhx.java;

import org.junit.Test;

import java.util.ArrayList;
import java.util.List;

public class MaxProfit {

    /**
     * 思路
     * 相邻数据相减，求都是正数的和
     */
    @Test
    public void test01() {
        int[] prices = new int[]{5, 4, 2, 6, 7, 8, 2, 1, 0, 10, 200, 1, 10, 300, 400};

        List<Integer> diffs = new ArrayList<>();
        int cur = prices[0];
        //求出相邻数据差值
        for (int i = 1; i < prices.length; i++) {
            diffs.add(prices[i] - cur);
            cur = prices[i];
        }
        Integer max = 0;
        Integer temMax = 0;
        for (int i = 0; i < diffs.size(); i++) {
            Integer diff = diffs.get(i);
            if (diff <= 0) {
                if (max < temMax) {
                    max = temMax;
                }
                temMax = 0;
                continue;
            }
            //差值大于0 证明递增
            temMax += diff;

        }
        if (max < temMax) {
            max = temMax;
        }
        System.out.println(max);

    }


    /**
     * 做一个升降序的标志
     * 升降序状态变化的时候就是拐点
     */
    @Test
    public void test02() {
        int[] prices = new int[]{5, 4, 2, 6, 7, 8, 2, 1, 0, 10, 200, 1, 10, 300, 400};

        int min = prices[0];
        int max = prices[0];
        int flag = 0; //0 升序 1降序
        int maxRange = 0;
        int temMaxRange = 0;
        for (int i = 1; i < prices.length; i++) {
            int cur = prices[i];
            int last = prices[i - 1];
            //初始化开始升/降序
            if (i == 1) {
                //升
                if (cur > min) {
                    flag = 0;
                } else {
                    flag = 1;
                }
            }

            //升
            if (cur > last) {
                //原来就是升序
                if (flag == 0) {
                    max = cur;
                } else {//拐点 最低点
                    min = last;
                    max = cur;
                    flag = 0;
                }
            } else {//降
                //原来就是降序
                if (flag == 1) {
                    min = cur;
                } else { //拐点 最高点
                    max = last;
                    flag = 1;
                    temMaxRange = max - min;
                    if (maxRange < temMaxRange) {
                        maxRange = temMaxRange;
                    }
                }
            }

        }
        //最后如果是升序，没有拐点
        temMaxRange = max - min;
        if (maxRange < temMaxRange) {
            maxRange = temMaxRange;
        }

        System.out.println(maxRange);

    }
}
```
